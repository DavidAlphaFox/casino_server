%- 夺宝协议定义：
%-     分类号:13
%-     描述: 夺宝
%- 返回码定义：
%-     0: 未定义错误  1：成功

%------------------------------------------
%Protocol: 14001 获取连环夺宝信息
%------------------------------------------
c >> s:
s >> c:
	int8: level  			关卡:1,2,3,4(龙珠探宝)
	int8: left_brick		剩余金砖数(第四关龙珠探宝忽略)
	int64: total_coins		游戏总分数
	int64: game_coins		本场游戏分数
	int64: pool_coins		累计奖池
	int:32 min_bet			单线最小押注数
	int:32 max_bet			单线最大押注数
	array(
		int:8 				压线条数(1~5)
		)

%------------------------------------------
%Protocol: 14002 下注
%------------------------------------------
c >> s:
	int8: bet_lines			押注条数
	int32: bet 				单线分数
s >> c:
	int8: error_code		错误码(0:成功) 
	string: error_str		错误消息提示
	int64: reward_coins		中奖金币数量 
	array
	(
		array
		(
			int:8 stone_id
		)					generate_stone_array 生成的石头
		array
		(
			int:8 stone_id
		)					dismiss_stone_array 消除的石头

	)						中奖过程展示

%------------------------------------------
%Protocol: 14003 消除动画播放完毕
%------------------------------------------
c >> s:

%------------------------------------------
%Protocol: 14004 奖池刷新
%------------------------------------------
s >> c:
	int:64 pool_coins

%------------------------------------------
%Protocol: 14005 重置关卡
%------------------------------------------
c >> s:
s >> c:
	int:8 error_code 0：成功

%------------------------------------------
%Protocol: 14006 获取中奖配置信息
%------------------------------------------
c >> s:
s >> c:
	array
	(
		int:8 level
		array
		(
			int:8 stone_id,		石头ID
			int:8 line_num,		相连多少个相消
			int:32 rate,		消除后奖励倍率
		)
	)
